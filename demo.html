Auth Widget Demo

A beautiful, standalone authentication modal that works anywhere. Just include the script and call the API.

Usage Example

// 1. Include the script
<script src="dist/auth-widget.js"></script>

// 2. Configure callbacks (optional)
window.AuthModalWidget.configure({
  onSignIn: (data) => {
    console.log('Sign in:', data.email);
  },
  onSignUp: (data) => {
    console.log('Sign up:', data.email, data.name);
  },
  onSocialLogin: (provider) => {
    console.log('Social login:', provider);
  }
});

// 3. Open modals
window.AuthModalWidget.openSignIn();
window.AuthModalWidget.openSignUp();
window.AuthModalWidget.close();
